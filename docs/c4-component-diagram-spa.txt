        spa -> analytics_bc     "Routes to /analytics"    "Angular Router"
        spa -> dashboard_bc     "Routes to /dashboard"    "Angular Router"
        spa -> maintenance_bc   "Routes to /maintenance"  "Angular Router"
        spa -> configuration_bc "Routes to /settings"      "Angular Router"
        spa -> reservation_bc   "Routes to /reservations" "Angular Router"
